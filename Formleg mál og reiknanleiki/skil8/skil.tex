\begin{lstlisting}
Input string w

while true do
	Sweep left to right across the tape, cross one a , b and c
	if single a left
		return accept
	else if single b left
		return accept
	else if single c left
		return accept
	else if a equals b or b equals c or c equals a
		return reject
